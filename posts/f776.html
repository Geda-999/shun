十一、密码验证程式

这种是无限次数

var password = 123456;

 var input;

   input = prompt("请输入密码");

 }

这种是有限次数

var password = 123456;

 var input;

 var entry_count = 0;

 var entry_limit = 3;

 var out_0f_limit = false;



 while(password != input && !out_0f_limit){

   entry_count++;

   if(entry_count<=entry_limit){

​     input = prompt("请输入密码");

   }

   else{

​     out_0f_limit = true;

   }

 }



 if(out_0f_limit){

   alert("登入成功");
