判断语句、条件语句、错误捕捉语句以及关键字

1、运算符

=   赋值
==  判断大小
=== 全等于，不仅判断大小，还判断类型

!   非
!=  不等于
!== 不全等于

&&  与
||  或

>=  大于等于
<=  小于等于

a++  累加1，a++相当于a = a + 1，先得出值，再运算
++a  累加1，++a相当于a = a + 1，先运算，再得出值
a+=  累加自身，a+=2相当于a = a + 2

--  累减1
-=  累减自身

逻辑【与】

表达式1 && 表达2

// 如果表达式1为真，则返回表达式2
// 如果表达式1为假，则返回表达式1
console.log(100 && 200);//true 200
console.log(0 && 200);//false 0


// 两边都是true才会返回true，否则返回错误false
var b1 = true && true;  //true
var b1 = true && false; //false

逻辑【或】

表达式1 || 表达式2

// 如果表达式1为真，则返回表达式1
// 如果表达式1为假，则返回表达式2
console.log(100 || 200);//true 100
console.log(0 || 200);//false 200


// 只要其中一边是true正确，就会返回true正确，
// 两边都是false错误，才返回false错误
var b2 = true || true;  //true
var b2 = true || false; //true
var b2 = false || false; //false

逻辑【非】

！表达1

// 如果表达1是真的，那么返回假（false）;
// 如果表达1是假的，那么返回真（true）;
console.log(!0);//true
console.log(!22);//false



// true正确会变成false错误的，
// false错误的会变成true正确的
var b3 = false;
console.log(!b3);//true
console.log(b3);//false

var b4 = true;
console.log(!b4);//false
console.log(b4);//true

好多【===】

// 1= 赋值  
// 2==判断是否等于  只判断值，不判断类型  
// 3=== 全等于  即判断，又判断类型
// != 不等于
//!== 完全不等于

// 2==判断是否等于  只判断值，不判断类型 
var b5 = 1 == '1';
console.log(b5); //true

// 3=== 全等于  即判断，又判断类型
var b5 = 1 === '1';
console.log(b5); //false

//!== 完全不等于
var b5 = 1 !== '1';
console.log(b5); //false

好多【++–】

// 加加在后 先计算，后自增1 a++

// 加加在前 先自身加1，后计算 ++a

var a = 100;
console.log(a++); //100

var a = 100;
console.log(++a); //101

var a = 100;
document.write(--a); //99

var a = 100;
document.write(a--); //100


//全写法（老实人写法）
d = d + 1;
//简写法（累加）
d += 4;
console.log(d); //5

//简写法（累减）
d -= 1;
console.log(d); //0

更多【++ –】

// 加加在后 先计算，后自增1 a++

// 加加在前 先自身加1，后计算 ++a

var a = 1;
var b = 100;
console.log(a++ + b++); //101
console.log(--a + b++); //100
console.log(a-- + --b); //100

判断语句switch

var a = prompt("请输入内容");
    switch (a * 1) {
        case 1:
            console.log("苹果");
            break;
        case 2:
            console.log("雪梨");
            break;
        case 3:
            console.log("我是水果哦！");
            break;
    }

let a = 1
switch (a){
    case 1:
    console.log("a是1")
        break;
    case 2:
        console.log("a是2")
        break;
    default:
        console.log("a不是1也不是2")
        break;
}

//如果要在switch case里面返回值，需要把break改为return
function fn() {
  let a = 1;
  switch (a) {
    case 1: {
      console.log("a是1");
      return;
    }
    case 2:
      console.log("a是2");
      break;
    default:
      console.log("a不是1也不是2");
      break;
  }
}

3、循环语句

for(){}

//1-10
for(let i = 1;i<=10;i++){
    console.log(i)
}

//10-1
for(let i = 10;i>=1;i--){
    console.log(i)
}

//99乘法表
document.write('<table border="1" cellspacing="0" cellpadding="10">');
for (let i = 1; i <= 9; i++) {
  document.write("<tr>");
  for (let j = 1; j <= i; j++) {
    document.write("<td>");
    document.write(i + "*" + j + "=" + i * j);
    document.write("</td>");
  }
  document.write("</tr>");
}
document.write("</table>");

4、错误捕捉语句

try catch

try {
  let a = 1;
  console.log(a);//成功
} catch (error) {
  console.log(error);//失败，错误对象
}

finally 最终，不管成功还是失败都执行

try {
  let a = 1;
  console.log(a);
} catch (error) {
  console.log(error);
} finally {
  console.log("不管成功还是失败，都执行");
}

throw

try {
  console.log(a);
} catch (error) {
  if(error.name === "ReferenceError"){
      throw "这是一个引用错误"
  }
} finally {
  console.log("不管成功还是失败，都执行");
}
